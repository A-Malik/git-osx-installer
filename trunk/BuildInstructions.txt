curl -O http://kernel.org/pub/software/scm/git/git-1.5.4.4.tar.bz2
tar jxvf git-1.5.4.4.tar.bz2
cd git-1.5.4.4
make prefix=/usr/local/git all
make prefix=/usr/local/git strip
sudo make prefix=/usr/local/git install
cd ..

# change hardlinks for symlinks
ruby #{InstallerFilesRoot}/UserScripts/symlink_git_hardlinks.rb

curl -O http://www.kernel.org/pub/software/scm/git/git-manpages-1.5.4.4.tar.bz2
mkdir -p /usr/local/git/man
sudo tar xjv -C /usr/local/git/man -f git-manpages-1.5.4.4.tar.bz2
